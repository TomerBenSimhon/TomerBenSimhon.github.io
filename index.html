Modular Platform System
An interface-driven set of platform types — boost, refuel, slippery, and death — so each platform only needs to implement a shared behavior contract. New platform types can be added without touching the controller. 
public interface IPlatformEffect
{
    void Apply(PlayerController player, Rigidbody rb, PlatformDetection runner, ref Coroutine coroutine);
    void Remove(PlayerController player, PlatformDetection runner, ref Coroutine coroutine);
}

// --- PlatformDetection.cs ---

void HandlePlatformEnter()
{
    if (!TryDetectPlatform(out RaycastHit hit)) return;
    if (!hit.transform.TryGetComponent(out IPlatformEffect newEffect)) return;

    TryRemoveEffect(_currentPlatform);

    _currentPlatform = hit.collider.gameObject;
    newEffect.Apply(_controller, _rb, this, ref _currentPlatformCoroutine);
    _isOnSpecialPlatform = true;
}

void HandlePlatformExit()
{
    if (!_isOnSpecialPlatform) return;
    if (IsStillOnPlatform()) return;

    TryRemoveEffect(_currentPlatform);
    _currentPlatform = null;
    _isOnSpecialPlatform = false;
}
public class BoostPlatform : MonoBehaviour, IPlatformEffect
{
    // ...tunable fields omitted: boostSpeed, boostDuration,
    // initialImpulseModifier, boostStackCount, springModifier, jumpModifier
    // (all [Range]-clamped in the Inspector)

    public void Apply(PlayerController player, Rigidbody rb, PlatformDetection runner, ref Coroutine boostCoroutine)
    {
        runner.SetPlatform(PlatformType.Boost);

        if (boostCoroutine != null)
        {
            runner.StopCoroutine(boostCoroutine);
            boostCoroutine = null;
        }

        Vector3 boost = rb.linearVelocity;
        boost.z = Mathf.Clamp(rb.linearVelocity.z * initialImpulseModifier, 0f,
            player.DefaultSettings.forwardSpeed * initialImpulseModifier * boostStackCount);
        rb.linearVelocity = boost.z > rb.linearVelocity.z ? boost : rb.linearVelocity;

        player.RuntimeSettings.forwardSpeed = boostSpeed;
        player.RuntimeSettings.groundSpringStrength = springModifier * player.DefaultSettings.groundSpringStrength;
        player.RuntimeSettings.jumpHeight = jumpModifier * player.DefaultSettings.jumpHeight;
    }

    public void Remove(PlayerController player, PlatformDetection runner, ref Coroutine boostCoroutine)
    {
        boostCoroutine = runner.StartCoroutine(StopBoost(player, boostDuration, runner));
    }

    IEnumerator StopBoost(PlayerController player, float duration, PlatformDetection runner)
    {
        yield return new WaitForSeconds(duration);

        // restores forwardSpeed, groundSpringStrength, and jumpHeight
        // back to their defaults — the mirror image of Apply() above
        player.RuntimeSettings.forwardSpeed = player.DefaultSettings.forwardSpeed;

        if (runner.CurrentPlatformType == PlatformType.Boost)
            runner.SetPlatform(PlatformType.None);
    }
}
