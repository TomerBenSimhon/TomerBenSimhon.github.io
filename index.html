void HandlePlatformEnter()
{
    if (!TryDetectPlatform(out RaycastHit hit)) return;
    if (!hit.transform.TryGetComponent(out IPlatformEffect newEffect)) return;

    TryRemoveEffect(_currentPlatform);

    _currentPlatform = hit.collider.gameObject;
    newEffect.Apply(_controller, _rb, this, ref _currentPlatformCoroutine);
    _isOnSpecialPlatform = true;
}

{
    if (!_isOnSpecialPlatform) return;
    if (IsStillOnPlatform()) return;

    TryRemoveEffect(_currentPlatform);
    _currentPlatform = null;
    _isOnSpecialPlatform = false;
}
public class BoostPlatform : MonoBehaviour, IPlatformEffect
{
    // ...tunable fields omitted: boostSpeed, boostDuration,
    // initialImpulseModifier, boostStackCount, springModifier, jumpModifier
    // (all [Range]-clamped in the Inspector)

    public void Apply(PlayerController player, Rigidbody rb, PlatformDetection runner, ref Coroutine boostCoroutine)
    {
        runner.SetPlatform(PlatformType.Boost);

        if (boostCoroutine != null)
        {
            runner.StopCoroutine(boostCoroutine);
            boostCoroutine = null;
        }

        Vector3 boost = rb.linearVelocity;
        boost.z = Mathf.Clamp(rb.linearVelocity.z * initialImpulseModifier, 0f,
            player.DefaultSettings.forwardSpeed * initialImpulseModifier * boostStackCount);
        rb.linearVelocity = boost.z > rb.linearVelocity.z ? boost : rb.linearVelocity;

        player.RuntimeSettings.forwardSpeed = boostSpeed;
        player.RuntimeSettings.groundSpringStrength = springModifier * player.DefaultSettings.groundSpringStrength;
        player.RuntimeSettings.jumpHeight = jumpModifier * player.DefaultSettings.jumpHeight;
    }

    public void Remove(PlayerController player, PlatformDetection runner, ref Coroutine boostCoroutine)
    {
        boostCoroutine = runner.StartCoroutine(StopBoost(player, boostDuration, runner));
    }

    IEnumerator StopBoost(PlayerController player, float duration, PlatformDetection runner)
    {
        yield return new WaitForSeconds(duration);

        // restores forwardSpeed, groundSpringStrength, and jumpHeight
        // back to their defaults — the mirror image of Apply() above
        player.RuntimeSettings.forwardSpeed = player.DefaultSettings.forwardSpeed;

        if (runner.CurrentPlatformType == PlatformType.Boost)
            runner.SetPlatform(PlatformType.None);
    }
}

[InitializeOnLoad]
public static class PrefabBrush
{
    static readonly Vector3 gridSize = new(3f, 0.5f, 3f);

    static PrefabBrush()
    {
        SceneView.duringSceneGui += OnSceneGUI;
        EnvironmentPrefabWindow.OnWindowSceneGUI += OnSceneGUI;
    }

    static void OnSceneGUI(SceneView sceneView)
    {
        if (!EnvironmentPrefabWindow.IsWindowOpen) { DestroyGhost(false); return; }

        // route raw editor input to the active tool (brush / rectangle)
        switch (_tool)
        {
            case BrushTool.Brush:     HandleBrushToolInput();  break;
            case BrushTool.Rectangle: HandleRectToolInput();   break;
        }
    }

    static Vector3 SnapToGrid(Vector3 pos, Vector3 grid) => new(
        Mathf.Round(pos.x / grid.x) * grid.x,
        Mathf.Round(pos.y / grid.y) * grid.y,
        Mathf.Round(pos.z / grid.z) * grid.z
    );

    // Snaps to the grid — but if that cell is already occupied, steps one
    // cell along the surface normal instead. That's what lets pieces be
    // painted onto the tops and sides of existing ones, building in 3D.
    static Vector3 SnapToFreePlace(Vector3 pos, Vector3 normal, Vector3 grid)
    {
        Vector3 snapped = SnapToGrid(pos, grid);

        if (!Physics.CheckBox(snapped, grid * 0.4f, Quaternion.identity,
                _paintingLayer, QueryTriggerInteraction.Collide))
            return snapped;

        // occupied → offset one cell along the dominant axis of the normal
        Vector3 dir = KeepLargestComponent(normal).normalized;
        return snapped + Vector3.Scale(grid, dir);
    }

    // Reduces a normal to its single strongest axis, so snapping stays
    // stable on flat surfaces instead of jittering between axes.
    static Vector3 KeepLargestComponent(Vector3 v)
    {
        float max = Mathf.Max(Mathf.Abs(v.x), Mathf.Abs(v.y), Mathf.Abs(v.z));
        return new Vector3(
            Mathf.Approximately(Mathf.Abs(v.x), max) ? v.x : 0f,
            Mathf.Approximately(Mathf.Abs(v.y), max) ? v.y : 0f,
            Mathf.Approximately(Mathf.Abs(v.z), max) ? v.z : 0f
        );
    }
}
Event-Driven Effect System
A data-driven effect system that lets designers wire up reactive camera, particle, and sound effects entirely in the Inspector — no new code per effect. An Effects component on any object listens to a central bitflag event bus, and each configured spec declares which events trigger or cancel it, with an optional per-sender filter. Camera effects are the standout: a manager sums every active effect's FOV, position, and rotation deltas each frame and feeds one combined result into Cinemachine, so multiple shakes and pulses stack additively instead of overriding each other. Adding a whole new reactive effect is pure Inspector setup; only a brand-new kind of camera effect needs code, via a small interface.
particle effects, triggered by stackable events
camera effects, additive across multiple effects
[DefaultExecutionOrder(-10)]
public class Effects : MonoBehaviour
{
    [SerializeReference] private List<CameraEffectSpec> cameraEffects = new();
    [SerializeField] private List<ParticleEffectSpec> particleEffects = new();
    [SerializeField] private List<SoundEffectSpec> soundEffects = new();

    private void OnEnable()  => GlobalEvents.Raised += OnEvent;
    private void OnDisable() => GlobalEvents.Raised -= OnEvent;

    // Every spec list follows the same cancel-then-trigger shape: first
    // let any spec that matches the cancel mask stop itself, then let
    // any spec that matches the trigger mask fire. Camera, particles,
    // and sound (via AudioManager) all go through this same pattern —
    // sound omitted below since it's identical to particles.
    private void OnEvent(GlobalEvents.Id id, GameObject sender)
    {
        // CAMERA
        if (CameraEffectsManager.I != null && cameraEffects != null)
        {
            foreach (var spec in cameraEffects)
                if (spec != null && spec.MatchesCancel(id, sender))
                    CameraEffectsManager.I.CancelTag(spec.tag);

            var call = new EffectCall { source = transform, target = null, position = transform.position, magnitude = 1f };

            foreach (var spec in cameraEffects)
            {
                if (spec == null || !spec.MatchesTrigger(id, sender)) continue;
                var eff = spec.Build(call);
                if (eff != null) CameraEffectsManager.I.Play(eff, spec.replaceExisting);
            }
        }

        // PARTICLES
        if (particleEffects != null)
        {
            foreach (var pspec in particleEffects)
                if (pspec != null && pspec.MatchesCancel(id, sender))
                {
                    pspec.Stop(id);
                    if (pspec.particleObject) pspec.particleObject.SetActive(false);
                }

            foreach (var pspec in particleEffects)
                if (pspec != null && pspec.MatchesTrigger(id, sender))
                {
                    if (pspec.particleObject) pspec.particleObject.SetActive(true);
                    pspec.Play();
                }
        }

        // ...SOUNDS follows the identical cancel → trigger shape, routed
        // through AudioManager instead of a ParticleController.
    }
}

{
    public float   fovAdd;
    public Vector3 posAdd;
    public Vector3 eulerAdd;

    public static CamDelta Zero => new CamDelta { fovAdd = 0f, posAdd = default, eulerAdd = default };
}

public interface ICameraEffect
{

    bool IsFinished { get; }
    void OnStart(CameraEffectsManager ctx);
    CamDelta Tick(float dt);
    void Cancel();
}

// --- CameraEffectsManager.cs ---
// Every active effect (a shake, a FOV pulse, etc.) is ticked every frame
// and its delta summed into one combined value — this is what lets
// several effects stack additively instead of one overriding another.
private readonly List<ICameraEffect> _effects = new();

private void LateUpdate()
{
    if (!vcam || !shakeExt) return;

    float dt = Time.unscaledDeltaTime;

    float   fovAdd = 0f;
    Vector3 posAdd = Vector3.zero;
    Vector3 eulAdd = Vector3.zero;

    for (int i = _effects.Count - 1; i >= 0; --i)
    {
        var e = _effects[i];
        CamDelta d = e.Tick(dt);

        fovAdd += d.fovAdd;
        posAdd += d.posAdd;
        eulAdd += d.eulerAdd;

        if (e.IsFinished)
            _effects.RemoveAt(i);
    }

    // Hand off the single combined result to Cinemachine, applied once
    // at the end of its pipeline (see CameraShakeExtension).
    shakeExt.posAdd   = applyPositionShake ? posAdd : Vector3.zero;
    shakeExt.eulerAdd = applyRotationShake ? eulAdd : Vector3.zero;
    shakeExt.fovAdd   = applyFovShake      ? fovAdd : 0f;
}

[field: SerializeField] public EnemyState CurrentState { get; private set; }

void Awake()
{
    // finds every state component on this enemy automatically —
    // no manual wiring, works no matter how many states exist
    allStates = new List<IEnemyState>(GetComponentsInChildren<IEnemyState>());
    ChangeState(EnemyState.Patrol);
}

// What actually decides when to switch states — the enemy notices the
// player, or loses track of them for long enough, and calls ChangeState.
void StateControl()
{
    if (CurrentState == EnemyState.Patrol && IsPlayerSpotted())
        ChangeState(EnemyState.Shocked);

    if (CurrentState == EnemyState.Chase && !CanSeePlayer() && chaseExitTimerRoutine == null)
        chaseExitTimerRoutine = StartCoroutine(ChaseStateExitTimer());
}

// Enables the matching state component and disables the rest.
// Works for any number of states without a switch statement to maintain.
public void ChangeState(EnemyState newState)
{
    if (newState == CurrentState) return;

    foreach (var state in allStates)
    {
        if (state.StateId == newState) state.Enter();
        else state.Exit();
    }

    CurrentState = newState;
}

Car — Tire Physics & Boost
Each tire runs its own raycast suspension and grip curve, with steering and static tires tuned separately and grip further scaled by normal load and blended toward a dedicated handbrake value for drifting. In the air, the car forward-simulates its own trajectory a few steps ahead to predict the surface it's about to land on, torquing itself toward that orientation before impact instead of slamming down flat. Boost runs on a fuel system that recharges correctly across controller switches, so sitting in Robot mode for a while comes back to a partially refueled tank, not an empty one.
climbing, launching, and self-leveling over rough terrain
private void ApplyFrictionPerTiresType(Tire[] tires, AnimationCurve frictionCurve)
{
    foreach (Tire tire in tires)
    {
        if (!tire.isGroundedExtended)
        {
            tire.ResetGrip(frictionCurve);
            continue;
        }

        // Lateral (sideways) velocity at the contact point — this is what
        // friction fights. Longitudinal roll is left untouched.
        Vector3 tireVel = _playerRb.GetPointVelocity(tire.tireTransform.position);
        float steeringVel = Vector3.Dot(tire.tireTransform.right, tireVel);

        // Grip factor stacks several modifiers: base slip curve, then the
        // partial-ground/airborne factor, the high-speed slide factor, and
        // finally a blend toward the low handbrake grip while braking.
        tire.gripFactor = CalcCurrentTireGripFactor(tire, frictionCurve, tireVel);
        tire.gripFactor *= _airborneFactor;
        tire.gripFactor *= _slippingFactor;
        tire.gripFactor = Mathf.Lerp(tire.gripFactor, settings.HandBrakeGrip, _handbrake01);

        // Acceleration needed to cancel that lateral velocity this frame,
        // scaled by grip. gripFactor = 1 kills all sideways slide; lower
        // values let the tire wash out.
        float desiredVelChange = -steeringVel * tire.gripFactor;
        float desiredAccel = desiredVelChange / Time.fixedDeltaTime;

        // While not handbraking, enforce a minimum bite so tires don't feel
        // greasy at low grip — but never more than what exactly stops the
        // slide, so it can't overshoot and snap the car the other way.
        if (!_carInput.HandBreakHeld)
        {
            float accelSign = Mathf.Sign(-steeringVel);
            float exactStopAccel = Mathf.Abs(steeringVel) / Time.fixedDeltaTime;
            float accelMagnitude = Mathf.Max(Mathf.Abs(desiredAccel), settings.MinLateralFrictionAccel);
            accelMagnitude = Mathf.Min(accelMagnitude, exactStopAccel);

            desiredAccel = accelSign * accelMagnitude;
        }

        _playerRb.AddForceAtPosition(
            tire.tireTransform.right * (_playerRb.mass / _allTires.Count * desiredAccel),
            tire.tireTransform.position,
            ForceMode.Force
        );
    }
}

// The base grip for one tire: how far sideways it's sliding (0..1) sampled
// through the friction curve, scaled by a normal-load factor, then eased
// toward its target so grip changes don't snap frame-to-frame (except on
// the exact frame the tire lands, where it's applied instantly).
private float CalcCurrentTireGripFactor(Tire tire, AnimationCurve frictionCurve, Vector3 tireVel)
{
    Vector3 carVel = Vector3.ProjectOnPlane(_playerRb.linearVelocity, _groundPlaneNormal);
    if (carVel.magnitude < 0.3f)
        return frictionCurve.Evaluate(0f);

    Vector3 tireVelAlongGround = Vector3.ProjectOnPlane(tireVel, tire.extendedGroundHit.normal);
    tire.slippingAmount = Mathf.Clamp(
        Vector3.Dot(tire.tireTransform.right, tireVelAlongGround.normalized), -1f, 1f);

    float gripFactor = frictionCurve.Evaluate(Mathf.Abs(tire.slippingAmount));
    gripFactor = Mathf.Clamp(gripFactor * GetNormalForceGripFactor(tire, tireVel.magnitude), frictionCurve.Evaluate(1f), 1);

    bool hasLanded = tire.isGroundedExtended && !tire.isGroundedLastFixedFrame;
    tire.gripFactor = hasLanded ? gripFactor :
        Mathf.MoveTowards(tire.gripFactor, gripFactor, 2f * Time.fixedDeltaTime);
    return tire.gripFactor;
}
private void AirAlignTorque()
{
    Vector3 targetUp = GetPredictedLandingNormal();
    Quaternion toTarget = Quaternion.FromToRotation(transform.up, targetUp);

    toTarget.ToAngleAxis(out float angleDeg, out Vector3 axisWorld);
    if (angleDeg > 180f) angleDeg -= 360f;

    Vector3 errorWorld = axisWorld * (angleDeg * Mathf.Deg2Rad);
    Vector3 errorLocal = transform.InverseTransformDirection(errorWorld);
    errorLocal.y = 0f; // keep yaw free for steering

    Vector3 angVelLocal = transform.InverseTransformDirection(_playerRb.angularVelocity);
    angVelLocal.y = 0f;

    Vector3 torqueLocal = (errorLocal * settings.AirAlignmentStrength) - (angVelLocal * settings.AirAlignmentDamping);
    _playerRb.AddTorque(transform.TransformDirection(torqueLocal), ForceMode.Acceleration);
}

// Forward-simulates gravity a few steps ahead to find the surface the car
// is about to land on, so AirAlignTorque can start correcting toward it
// before impact instead of reacting after touchdown. Cached for a few
// fixed frames since raycasting every step for a value that barely
// changes frame-to-frame is wasted work.
private Vector3 GetPredictedLandingNormal()
{
    if (_landingNormalCounter > 0)
    {
        _landingNormalCounter--;
        return _cachedLandingNormal;
    }
    _landingNormalCounter = LandingNormalUpdateInterval;

    Vector3 pos = transform.position;
    Vector3 vel = _playerRb.linearVelocity;
    float dt = 0.1f;

    for (int i = 0; i < 50; i++)
    {
        float dv = -settings.TerminalVelocity - vel.y;
        float gravityForce = Mathf.Clamp(dv * GravityControlFactor, -settings.GravityStrength, Mathf.Infinity);
        vel.y += gravityForce * dt;

        Vector3 nextPos = pos + vel * dt;
        Vector3 step = nextPos - pos;

        if (Physics.Raycast(pos, step.normalized, out RaycastHit hit, step.magnitude + 0.1f, _groundLayer, QueryTriggerInteraction.Ignore))
        {
            _cachedLandingNormal = hit.normal;
            return _cachedLandingNormal;
        }
        pos = nextPos;
    }

    _cachedLandingNormal = _groundPlaneNormal;
    return _cachedLandingNormal;
}

Robot — Physics-Based Locomotion
Movement is applied as a force at an offset point below center of mass rather than a direct velocity set, with a spring-damper suspension holding it at ride height and a slope-walkability check that switches to a slide state on anything too steep to stand on. Rotation is entirely torque-driven: an upright spring keeps it standing, and a separate camera-relative yaw torque turns it to face wherever the camera's looking, independent of movement direction — so it can strafe.
climbing slopes and canyon terrain
private void UpdateRotation()
{
    ApplyUprightSpringTorque();
    ApplyCameraYawTorqueIfValid();
}

// A spring-damper torque toward world-up: the cross product of current up
// and world-up gives the correction axis, damped by tilt angular velocity
// (GetTiltAngularVelocityWorld, omitted — just strips the yaw component)
// so it settles instead of oscillating.
private void ApplyUprightSpringTorque()
{
    Vector3 tiltAxisWorld = Vector3.Cross(_playerRb.transform.up, Vector3.up);
    Vector3 tiltAngVelWorld = GetTiltAngularVelocityWorld();

    Vector3 uprightTorque = (tiltAxisWorld * settings.UprightStrength) - (tiltAngVelWorld * settings.UprightDamping);
    _playerRb.AddTorque(uprightTorque, ForceMode.Acceleration);
}

// Turns the robot to face the camera's flattened forward, independent of
// which way it's actually moving — movement input and facing direction
// are fully decoupled, so it can strafe.
private void ApplyCameraYawTorqueIfValid()
{
    if (!TryGetFlattenedCameraForward(out Vector3 camForwardFlat)) return;

    Vector3 forwardFlat = GetFlattenedForwardWorld();
    float yawErrorRad = Vector3.SignedAngle(forwardFlat, camForwardFlat, Vector3.up) * Mathf.Deg2Rad;
    float yawVelLocal = transform.InverseTransformDirection(_playerRb.angularVelocity).y;

    float yawAccel = (yawErrorRad * settings.YawStrength) - (yawVelLocal * settings.YawDamping);
    _playerRb.AddRelativeTorque(0f, yawAccel, 0f, ForceMode.Acceleration);
}

Magnetic Push & Pull — Modular Magnetizable System
The core mechanic: the left hand pushes, the right hand pulls. A SphereCast widened into a magnetic "field" reaches out from the hand and finds a target — but the player script never touches that object's physics directly. It just finds a shared Magnetizable component and fires its push or pull event, handing over the hand transform, the hit point, and the range. What that means physically is entirely up to whatever's on the other end: a free rigidbody gets dragged and can be caught into a hover in front of the camera, a rail-bound platform slides along its track, and a fixed magnetic wall reels the platform you're standing on toward it instead. Same input, completely different response per object — and a brand-new magnetizable type is pure Inspector wiring, with zero changes to the player.
pulling a crate into a hover, then pushing it away
// A widened SphereCast from the hand acts as the magnetic "field". The
// player never manipulates the object directly — it finds a Magnetizable
// and fires its pull event. What that means physically is up to the
// object on the other end.
private void PullingLogic()
{
    Vector3 origin = _rightHandTransform.position - _cam.transform.forward * width;
    bool isHit = Physics.SphereCast(origin, width, _cam.transform.forward,
                                    out _pullingHit, pullDistance, _magnetizableLayer);

    if (isHit)
    {
        // If the beam slid onto a different object this frame, release the
        // previous one first — unless it's mid-hover and should be kept.
        var hit = _pullingHit.collider.GetComponent<Magnetizable>();
        if (_currentPulledMagnetizable != hit)
        {
            if (_currentPulledMagnetizable != null && !_currentPulledMagnetizable.isHovering)
            {
                ReleasePull();
                _currentPulledMagnetizable = hit;
            }
            else if (_currentPulledMagnetizable == null)
                _currentPulledMagnetizable = hit;
        }
    }

    bool isHover = _currentPulledMagnetizable != null && _currentPulledMagnetizable.isHovering;

    if (isHit || isHover)
    {
        PullingState = PullingState2.Pulling;
        _currentPulledMagnetizable.Pulled(_rightHandTransform, _pullingHit.point, pullDistance, gameObject);
    }
    else if (_currentPulledMagnetizable != null)
        ReleasePull();
    else
        PullingState = PullingState2.LookingForPull;
}

}
// One of several receivers wired to the pull event. Past a distance
// threshold it drags the body toward the hand; once near enough it flips
// into a hover that holds it floating in front of the camera.
public void OnPulled(Transform hand, Vector3 hitPoint, float pullRange, GameObject player)
{
    if (currentMovableObjectState == MovableObjectState.Pushed) return;

    float distance = Vector3.Distance(hand.position, hitPoint);
    currentMovableObjectState =
        distance > hoverDistance && currentMovableObjectState != MovableObjectState.Hover
            ? MovableObjectState.Pulled
            : MovableObjectState.Hover;

    switch (currentMovableObjectState)
    {
        case MovableObjectState.Pulled: MagnetToPlayer(hand, hitPoint, pullRange); break;
        case MovableObjectState.Hover:  _magnetizable.isHovering = true; HoverOnPlayer(hand); break;
    }
}

private void MagnetToPlayer(Transform hand, Vector3 hitPoint, float pullRange)
{
    _rb.useGravity = false;
    Vector3 pullDir = (hand.position - hitPoint).normalized;
    float distance = Vector3.Distance(hand.position, hitPoint);
    // Closer = stronger: force is mapped high→low across the range, then
    // squared so the pull accelerates hard as the object closes in.
    float force = CharacterControllerUtil.MapValue(distance, hoverDistance, pullRange, maxForce, minForce);
    _rb.AddForceAtPosition(force * force / weight * pullDir, hitPoint, ForceMode.Acceleration);
}

// Hover: lock to a point floating in front of the camera by driving
// velocity toward it, so the held object tracks wherever you look.
private void HoverOnPlayer(Transform hand)
{
    _rb.useGravity = false;
    Vector3 target = (hand.position + _camTransform.forward * hoverDistance) - transform.position;
    _rb.linearVelocity = target * hoverSmoothness;
}
Magnetic Jump — Self-Launch off Magnetic Surfaces
The magnet works both ways. Held in mid-air over a magnetic surface, the same repulsion that pushes objects away instead launches you upward. A downward SphereCast checks for a magnetic surface within range, and the closer it is, the harder the kick — force mapped by distance and squared, so a launch from right above a plate throws you far higher than one at the edge of range. It's gated behind a mid-air re-press of jump, so it reads as a deliberate second, magnetic boost rather than a free hover, and it feeds a separate "magnetic vertical velocity" channel that the controller lerps into — keeping normal gravity jumps and magnetic launches from fighting each other frame to frame.
re-pressing jump over a magnetic plate to launch upward
// Triggered by a mid-air re-press of jump (jumpCount >= 1), so it reads
// as a deliberate magnetic boost, not a free hover. A downward SphereCast
// looks for a magnetic surface within range...
private void MagnetJump()
{
    if (_pushAndPull.PushingState is PushingState2.Pushing or PushingState2.LookingForPush) return;

    Vector3 origin = transform.position + Vector3.up * (width + 0.1f);
    if (Physics.SphereCast(origin, width, Vector3.down, out RaycastHit hit, playerPushDistance, magneticJumpLayer))
        _pushAndPull.ChangePushingState(PushingState2.PushingPlayer);
    else
        _pushAndPull.ChangePushingState(PushingState2.LookingForPushPlayer);

    if (_pushAndPull.PushingState == PushingState2.PushingPlayer)
    {
        // ...the nearer the surface, the stronger the repulsion. Mapped
        // close→far and squared, so a launch from right above the plate
        // kicks far harder than one from the edge of range.
        float distance = Vector3.Distance(transform.position, hit.point);
        float force = CharacterControllerUtil.MapValue(distance, 0, playerPushDistance, 5f, 0f);
        _controller.AddMagneticVerticalVelocity(playerPushForce * force * force);
    }
}

// PlayerController.cs — while launching, vertical velocity is lerped toward
// the magnetic channel instead of just integrating gravity, so a magnetic
// boost and a normal fall don't overwrite each other.
_verticalVelocity = _pushAndPull.PushingState != PushingState2.PushingPlayer
    ? _verticalVelocity - gravity * Time.deltaTime
    : Mathf.Lerp(_verticalVelocity, _magneticVerticalVelocity, Time.deltaTime * 2f);
Physics Character Controller — Moving-Platform Momentum
A CharacterController-driven locomotion built from scratch: camera-relative acceleration with separate ground/air drag and speed, sphere-cast grounding, and steep-wall projection so you slide off anything past the slope limit instead of sticking to it. The interesting part is how it rides moving platforms — rather than parenting the player to the platform (which jitters and fights rotation), it tracks the platform's velocity as its own channel, adds it on top of input velocity, then bleeds that inherited momentum away with air/ground drag only once you leave. So you carry a platform's speed cleanly into a jump and land naturally. As a bonus cross-system touch, standing on a magnet-driven platform flags it as non-magnetizable, so you can't push the very platform you're riding out from under yourself.
riding a magnet-driven platform and carrying its momentum into a jump
// Moving platforms aren't parented to the player — that jitters and fights
// rotation. Their velocity is carried as its own channel, added on top of
// input velocity, and only bled off gradually once you leave, so momentum
// transfers into a jump instead of vanishing.
Vector3 moveDelta = lateralAcceleration * Time.deltaTime * moveDir;

// Subtract last frame's platform velocity so it isn't counted twice.
Vector3 newVelocity = _controller.velocity + moveDelta - _movingPlatformVelocity;

CheckForMovablePlatform();
_movingPlatformVelocity = CurrentMovablePlatform != null
    ? Vector3.MoveTowards(_movingPlatformVelocity, CurrentMovablePlatform.Velocity, Time.deltaTime * 100f)
    : isGrounded
        ? Vector3.MoveTowards(_movingPlatformVelocity, Vector3.zero, drag * Time.deltaTime)
        : Vector3.MoveTowards(_movingPlatformVelocity, Vector3.zero, inAirDrag * Time.deltaTime);

// Clamp only the input-driven part, then add the platform channel back, so
// standing still on a fast platform never gets clamped down to a crawl.
newVelocity = Vector3.ClampMagnitude(new Vector3(newVelocity.x, 0f, newVelocity.z), _lerpedLateralSpeed);
newVelocity.y += _verticalVelocity;
Vector3 finalVelocity = newVelocity + _movingPlatformVelocity;

_controller.Move(finalVelocity * Time.deltaTime);

// Grounding check specific to platforms: a short downward SphereCast on the
// platform layer. Entering one caches it (and flags it so it can't be
// pushed/pulled while ridden); leaving clears it so the bleed-off kicks in.
private void CheckForMovablePlatform()
{
    Vector3 origin = transform.position + Vector3.up * _controller.radius;
    bool onPlatform = Physics.SphereCast(origin, _controller.radius, Vector3.down,
                                         out RaycastHit hit, 0.2f, _movingPlatformLayer);

    if (onPlatform && _currentMovingPlatformObject != hit.collider.gameObject)
    {
        CurrentMovablePlatform = hit.transform.GetComponent<MovablePlatform>();
        if (hit.transform.TryGetComponent(out Magnetizable m))
            m.isPlayerOnMovablePlatform = true;
        _currentMovingPlatformObject = hit.collider.gameObject;
    }
    else if (!onPlatform && CurrentMovablePlatform != null)
    {
        if (CurrentMovablePlatform.TryGetComponent(out Magnetizable m))
            m.isPlayerOnMovablePlatform = false;
        CurrentMovablePlatform = null;
        _currentMovingPlatformObject = null;
    }
}
Ragdoll Death & Checkpoint Respawn
On death, the animated character hands off to a ragdoll. At startup it auto-collects every ragdoll-tagged collider and rigidbody in the hierarchy and disables them, so dying is cheap: switch off the animation rig and controller, switch on the ragdoll bodies, and pour the CharacterController's last velocity into them — so the corpse keeps the momentum that killed it instead of collapsing straight down. Checkpoints are set by simple trigger volumes, and the GameManager reloads the scene after a short beat and drops the player back at the last one. Loose magnetizable objects get their own respawn triggers, so a crate you knock into the void returns to its start position rather than staying lost.
a hazard kills the player into a ragdoll, then respawns at the checkpoint
// Ragdoll parts are gathered and disabled at startup (GetRagdollParts,
// omitted — it just collects every "Ragdoll"-tagged collider/rigidbody).
// On death the animated rig and controller switch off, the ragdoll switches
// on, and the CharacterController's velocity is poured into the bodies so
// the corpse carries the momentum that killed it.
void EnableRagdoll()
{
    Vector3 velocity = _characterController.velocity;

    _characterController.enabled = false;
    foreach (var c in _componentsToDisable) c.enabled = false;

    foreach (var col in _colliders) col.enabled = true;
    foreach (var rb in _rigidbodies)
    {
        rb.useGravity = true;
        rb.linearVelocity = velocity;   // inherit the killing blow's speed
        rb.angularVelocity = Vector3.zero;
    }
}

public void Die()
{
    EnableRagdoll();
    GameManager.Instance.ReloadScene(2f);   // a brief beat before respawn
}
// Checkpoint.cs — a trigger volume that records the last safe position.
private void OnTriggerEnter(Collider other)
{
    if (other.CompareTag("Player"))
        CheckpointManager.Instance.SetCheckpoint(transform.position);

// GameManager.cs — reload after a delay, then drop the player back at the
// stored checkpoint once the fresh scene finishes loading.
public void ReloadScene(float delay = 0f)
{
    StartCoroutine(ReloadSceneDelay(delay));
}

private IEnumerator ReloadSceneDelay(float delay)
{
    yield return new WaitForSeconds(delay);
    Destroy(_player);
    SceneManager.LoadScene(SceneManager.GetActiveScene().buildIndex);
}

private void OnSceneLoaded(Scene scene, LoadSceneMode mode)
{
    if (_player != null)
        _player.transform.position = CheckpointManager.Instance.CurrentCheckpoint;
}
